<script>


const WEB_APP_URL = https://script.google.com/macros/s/AKfycbz55-o4m_PTZTFk_B2dtZfexxEyy-w-ORv0mEgndBeNNEClGnQrxfv0Wc4lQZ6yMMH2/exec;


const video = document.getElementById("video");

const canvas = document.getElementById("canvas");

const result = document.getElementById("result");

const countdownEl = document.getElementById("countdown");


navigator.mediaDevices.getUserMedia({

  video:{facingMode:"user"}

}).then(stream=>{

  video.srcObject = stream;

});


function startCountdown(){

  let count = 3;

  countdownEl.innerText = count;


  const interval = setInterval(()=>{

    count--;

    if(count>0){

      countdownEl.innerText = count;

    }else{

      clearInterval(interval);

      countdownEl.innerText="";

      capture();

    }

  },1000);

}


function capture(){

  canvas.width = video.videoWidth;

  canvas.height = video.videoHeight;

  const ctx = canvas.getContext("2d");

  ctx.drawImage(video,0,0);

  generate4R(canvas.toDataURL("image/jpeg",1.0));

}


function generate4R(photoData){


  const layoutCanvas = document.createElement("canvas");

  const ctx = layoutCanvas.getContext("2d");


  layoutCanvas.width = 1800;

  layoutCanvas.height = 1200;


  ctx.fillStyle="#000";

  ctx.fillRect(0,0,1800,1200);


  ctx.fillStyle="gold";

  ctx.font="bold 80px Arial";

  ctx.textAlign="center";

  ctx.fillText("KEMENTERIAN HAJI DAN UMRAH",900,120);

  ctx.fillText("KABUPATEN MAGELANG",900,200);


  const img = new Image();

  img.src = photoData;


  img.onload = function(){

    ctx.drawImage(img,300,250,1200,750);


    ctx.font="40px Arial";

    ctx.fillText("Photo Booth Kantor",900,1100);


    const finalImage = layoutCanvas.toDataURL("image/jpeg",1.0);

    result.src = finalImage;


    uploadToDrive(finalImage);

  }

}


function uploadToDrive(imageData){


fetch(WEB_APP_URL,{

  method:"POST",

  body:JSON.stringify({image:imageData})

})

.then(res=>res.json())

.then(data=>{


  generateQR(data.url);


});

}


function generateQR(url){


  document.getElementById("qrSection").style.display="block";

  document.getElementById("qrcode").innerHTML="";


  new QRCode(document.getElementById("qrcode"),{

    text:url,

    width:220,

    height:220

  });

}
